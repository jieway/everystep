class Solution {
public:
    static bool cmp(vector<int> &a, vector<int> &b) {return a[1]<b[1];}
    int findLongestChain(vector<vector<int>>& pairs) {
        sort(pairs.begin(),pairs.end(), cmp);
        int len = pairs.size();
        int t =  pairs[0][1];
        int k = 1;
        for(int i = 1; i < len ;i ++) {
            if (pairs[i][0] > t) {
                k++;
                t = pairs[i][1];
            }
        }
        return k;
    }
};
