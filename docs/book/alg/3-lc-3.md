class Solution {
public:
    int lengthOfLongestSubstring(string s) {
        int m = 0, left = 0;
        unordered_map<char, int> hash;
        for (int i = 0; i < s.length(); i++) {
            if (hash.find(s[i]) != hash.end()) {
                left = max(left , hash[s[i]] + 1);
            }
            hash[s[i]] = i;
            m = max(m , i - left + 1);
        }
        return m;
    }
};
